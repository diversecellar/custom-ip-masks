# Git Commands to Push Custom IP Masks to GitHub
# ================================================

# 1. Navigate to your project directory
cd "C:\Users\kabwe\OneDrive - University of Cape Town\2025\Work\custom-ip-masks"

# 2. Initialize Git repository (if not already done)
git init

# 3. Add all files to Git
git add .

# 4. Create initial commit
git commit -m "Initial commit: Custom IP Masks Proxy Server

- Complete Flask-based HTTP/HTTPS proxy server
- IP masking and anonymization features
- Header sanitization and User-Agent rotation
- Rate limiting and security features
- Configuration management system
- Start/stop utilities and test suite
- Comprehensive documentation"

# 5. Create GitHub repository (using GitHub CLI - optional)
# If you have GitHub CLI installed:
# gh repo create custom-ip-masks --public --description "Custom HTTP/HTTPS proxy server for IP masking and anonymization"

# 6. Add GitHub remote (replace with your actual repo URL)
git remote add origin https://github.com/diversecellar/custom-ip-masks.git

# 7. Set main branch name
git branch -M main

# 8. Push to GitHub
git push -u origin main
